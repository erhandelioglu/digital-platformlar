// SÜRÜKLE BIRAK (DRAG & DROP) SIRALAMA İÇİN YARDIMCI DURUM YÖNETİCİSİ
class DragDropListState(
    val lazyListState: androidx.compose.foundation.lazy.LazyListState,
    private val onMove: (Int, Int) -> Unit
) {
    var draggedDistance by mutableStateOf(0f)
    var initiallyDraggedElement by mutableStateOf<androidx.compose.foundation.lazy.LazyListItemInfo?>(null)
    var currentIndexOfDraggedItem by mutableStateOf<Int?>(null)

    private val currentElement: androidx.compose.foundation.lazy.LazyListItemInfo?
        get() = initiallyDraggedElement?.let { init ->
            lazyListState.layoutInfo.visibleItemsInfo.find { it.index == currentIndexOfDraggedItem }
        }

    fun onDragStart(offset: androidx.compose.ui.geometry.Offset) {
        lazyListState.layoutInfo.visibleItemsInfo
            .find { item -> offset.y.toInt() in item.offset..(item.offset + item.size) }
            .also {
                initiallyDraggedElement = it
                currentIndexOfDraggedItem = it?.index
            }
    }

    fun onDragInterrupted() {
        initiallyDraggedElement = null
        currentIndexOfDraggedItem = null
        draggedDistance = 0f
    }

    fun onDrag(offset: androidx.compose.ui.geometry.Offset) {
        draggedDistance += offset.y
        val initialElement = initiallyDraggedElement ?: return
        val startOffset = initialElement.offset + draggedDistance

        val targetItem = lazyListState.layoutInfo.visibleItemsInfo.find { item ->
            startOffset.toInt() in item.offset..(item.offset + item.size) &&
                    item.index != currentIndexOfDraggedItem
        }

        if (targetItem != null) {
            currentIndexOfDraggedItem?.let { current ->
                onMove(current, targetItem.index)
            }
            currentIndexOfDraggedItem = targetItem.index
            draggedDistance += initialElement.offset - targetItem.offset
        }
    }

    fun elementOffset(index: Int): Float {
        if (index != currentIndexOfDraggedItem) return 0f
        val element = currentElement ?: return 0f
        return draggedDistance + (initiallyDraggedElement?.offset ?: 0) - element.offset
    }
}

@Composable
fun rememberDragDropListState(
    lazyListState: androidx.compose.foundation.lazy.LazyListState = rememberLazyListState(),
    onMove: (Int, Int) -> Unit
): DragDropListState {
    return remember { DragDropListState(lazyListState, onMove) }
}

    private fun initializePlatforms(jsonString: String) {
        try {
            val jsonArray = JSONArray(jsonString)
            val temporaryList = mutableListOf<DigitalPlatform>()

            for (i in 0 until jsonArray.length()) {
                val obj = jsonArray.getJSONObject(i)
                temporaryList.add(
                    DigitalPlatform(
                        name = obj.optString("name", "Bilinmeyen Platform"),
                        packageName = obj.optString("packageName", ""),
                        logoUrl = obj.optString("logoUrl", ""),
                        webUrl = obj.optString("webUrl", ""),
                        category = obj.optString("category", "Genel")
                    )
                )
            }

            digitalPlatformsList.clear()
            digitalPlatformsList.addAll(temporaryList)
        } catch (e: Exception) {
            e.printStackTrace()
        }
    }

    private fun openDigitalPlatform(platform: DigitalPlatform) {
        val seciliKanal = currentChannelState.value
        if (seciliKanal != null && !seciliKanal.isRadio) {
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                val params = PictureInPictureParams.Builder()
                    .setAspectRatio(Rational(16, 9))
                    .build()
                enterPictureInPictureMode(params)
            }
        }

        val pm = packageManager
        val intent = pm.getLaunchIntentForPackage(platform.packageName)
        if (intent != null) {
            startActivity(intent)
        } else {
            val isIOS = Build.MODEL.lowercase().contains("iphone") || Build.MODEL.lowercase().contains("ipad")
            try {
                if (isIOS && platform.appStoreUrl.isNotEmpty()) {
                    val appStoreIntent = android.content.Intent(android.content.Intent.ACTION_VIEW, Uri.parse(platform.appStoreUrl))
                    startActivity(appStoreIntent)
                } else {
                    val playStoreIntent = android.content.Intent(
                        android.content.Intent.ACTION_VIEW,
                        Uri.parse("market://details?id=${platform.packageName}")
                    ).apply { setPackage("com.android.vending") }
                    startActivity(playStoreIntent)
                }
            } catch (e: Exception) {
                try {
                    val searchUrl = if (isIOS) "https://apps.apple.com/tr/search?term=${Uri.encode(platform.name)}"
                    else "market://search?q=${Uri.encode(platform.name)}"

                    val searchIntent = android.content.Intent(android.content.Intent.ACTION_VIEW, Uri.parse(searchUrl))
                    if (!isIOS) searchIntent.setPackage("com.android.vending")
                    startActivity(searchIntent)
                } catch (ex: Exception) {
                    val finalWebIntent = android.content.Intent(android.content.Intent.ACTION_VIEW, Uri.parse(platform.webUrl))
                    startActivity(finalWebIntent)
                }
            }
        }
    }

    private fun initializePlayer() {
        if (exoPlayer != null) return
        val trustAllCerts = arrayOf<javax.net.ssl.TrustManager>(object : javax.net.ssl.X509TrustManager {
            override fun checkClientTrusted(chain: Array<out java.security.cert.X509Certificate>?, authType: String?) {}
            override fun checkServerTrusted(chain: Array<out java.security.cert.X509Certificate>?, authType: String?) {}
            override fun getAcceptedIssuers(): Array<java.security.cert.X509Certificate> = arrayOf()
        })
        val sslContext = javax.net.ssl.SSLContext.getInstance("SSL").apply { init(null, trustAllCerts, java.security.SecureRandom()) }
        val client = OkHttpClient.Builder().sslSocketFactory(sslContext.socketFactory, trustAllCerts[0] as javax.net.ssl.X509TrustManager).hostnameVerifier { _, _ -> true }.build()
        val dataSourceFactory = androidx.media3.datasource.okhttp.OkHttpDataSource.Factory(client).setUserAgent("Mozilla/5.0")

        exoPlayer = ExoPlayer.Builder(this).setMediaSourceFactory(androidx.media3.exoplayer.source.DefaultMediaSourceFactory(this).setDataSourceFactory(dataSourceFactory)).build().apply {
            playWhenReady = true
            addListener(object : Player.Listener {
                override fun onPlayerError(error: PlaybackException) {
                    handleChannelError()
                }
                override fun onIsPlayingChanged(isPlaying: Boolean) {
                    isPlayingState.value = isPlaying
                    if (isPlaying) window.addFlags(WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON)
                    else window.clearFlags(WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON)
                }
            })
        }
        sharedPlayerView = PlayerView(this).apply { player = exoPlayer; useController = false; layoutParams = FrameLayout.LayoutParams(-1, -1) }
    }

    private fun handleChannelError() {
        val currentChannel = currentChannelState.value ?: return

        if (currentChannel.currentUrlIndex < currentChannel.urls.size - 1) {
            currentChannel.currentUrlIndex++
            val nextBackupUrl = currentChannel.urls[currentChannel.currentUrlIndex]
            runOnUiThread {
                val cihazTuru = if (currentChannel.isRadio) "📻 Radyo" else "📺 TV"
                Toast.makeText(this, "$cihazTuru yedek linke geçiliyor (${currentChannel.currentUrlIndex + 1}/${currentChannel.urls.size})", Toast.LENGTH_SHORT).show()
                playChannel(nextBackupUrl)
            }
        } else {
            playNextChannelAvailable()
        }
    }

    private fun playChannel(url: String) {
        try {
            exoPlayer?.apply { stop(); clearMediaItems(); setMediaItem(MediaItem.fromUri(Uri.parse(url))); prepare(); play() }
        } catch (e: Exception) { e.printStackTrace() }
    }

    private fun closeCurrentChannel() {
        exoPlayer?.stop(); exoPlayer?.clearMediaItems()
        currentChannelState.value = null; isFullScreen.value = false
        requestedOrientation = ActivityInfo.SCREEN_ORIENTATION_UNSPECIFIED
        getSharedPreferences("aile_medya_prefs", MODE_PRIVATE).edit().remove("son_kanal_url").apply()
    }

    private fun playNextChannelAvailable() {
        if (filteredChannelsSnapshot.isNotEmpty() && currentFilteredIndex != -1) {
            runOnUiThread {
                currentFilteredIndex = (currentFilteredIndex + 1) % filteredChannelsSnapshot.size
                val next = filteredChannelsSnapshot[currentFilteredIndex]
                next.currentUrlIndex = 0
                currentChannelState.value = next; playChannel(next.urls[0])
            }
        }
    }

        isFullScreen.value = full
        requestedOrientation = if (full) ActivityInfo.SCREEN_ORIENTATION_LANDSCAPE else ActivityInfo.SCREEN_ORIENTATION_UNSPECIFIED
    }

    fun saveFavoritesToDisk() {
        // Siralamayi korumak icin Set yerine List seklinde manuel JSON String veya Virgülle ayrilmis sekilde kaydediyoruz
        val csvFormat = favoriteIdsState.joinToString(",")

    private fun loadFavoritesFromDisk() {
        val prefs = getSharedPreferences("aile_medya_prefs", MODE_PRIVATE)
        val csvFormat = prefs.getString("favoriler_sirali", null)

        favoriteIdsState.clear()
        if (!csvFormat.isNullOrEmpty()) {
            favoriteIdsState.addAll(csvFormat.split(","))
        } else {
            // Eski kayitlar varsa dönüsüm sağla
            prefs.getStringSet("favoriler", null)?.let {
                favoriteIdsState.addAll(it)
            }
        }
    }

    private fun saveLastChannel(channel: MediaChannel) {
        if (channel.urls.isEmpty()) return
        getSharedPreferences("aile_medya_prefs", MODE_PRIVATE).edit().apply {
            putString("son_kanal_id", channel.id)
            putString("son_kanal_title", channel.title)
            putString("son_kanal_url", channel.urls[0])
            putString("son_kanal_logo", channel.logoUrl)
            putBoolean("son_kanal_is_radio", channel.isRadio)
            putString("son_kanal_category", channel.category)
        }.apply()
    }

    private fun restoreLastChannel(channelsList: List<MediaChannel>) {
        val prefs = getSharedPreferences("aile_medya_prefs", MODE_PRIVATE)
        val url = prefs.getString("son_kanal_url", null)
        if (url != null) {
            val title = prefs.getString("son_kanal_title", "Canlı Yayın") ?: "Canlı Yayın"

            val matchedChannel = channelsList.find { it.title.uppercase() == title.uppercase() }
            if (matchedChannel != null) {
                matchedChannel.currentUrlIndex = 0
                currentChannelState.value = matchedChannel
                val matchIndex = channelsList.indexOf(matchedChannel)
                if (matchIndex != -1) currentFilteredIndex = matchIndex
                playChannel(matchedChannel.urls[0])
            }
        }
    }

    private suspend fun loadOnlineMediaList() {

        val ulkeKodu = currentCountryCodeState.value.lowercase()

            tvKaynaklari.add("https://raw.githubusercontent.com/suphero/IPTV/master/TR.m3u")

            tvKaynaklari.add("https://onureroz.com/indirmeler/turk/karma-index.m3u")

        }

        var internettenYuklendiMi = false
        val indirilenTumM3uIcerigi = StringBuilder()

        withContext(Dispatchers.IO) {

            try {
                // !!! GİTHUB RAW LİNKİNİ BURAYA EKLEMEYİ UNUTMA !!!
                val platformRequest = Request.Builder()
                    .url("https://raw.githubusercontent.com/erhandelioglu/digital-platformlar/refs/heads/main/platformlar.json")
                    .build()
                val platformResponse = client.newCall(platformRequest).execute()
                if (platformResponse.isSuccessful) {
                    val jsonIcerik = platformResponse.body?.string() ?: ""
                    if (jsonIcerik.isNotEmpty()) {
                        withContext(Dispatchers.Main) {
                            initializePlatforms(jsonIcerik)
                        }
                    }
                }
            } catch (e: Exception) {
                e.printStackTrace()
            }

            for (kaynakUrl in tvKaynaklari) {
                try {
                    val request = Request.Builder().url(kaynakUrl).build()
                    val response = client.newCall(request).execute()
                    if (!response.isSuccessful) continue

                    val bodyString = response.body?.string() ?: ""
                    if (bodyString.isNotEmpty() && (bodyString.contains("#EXTINF") || bodyString.contains("#EXTM3U"))) {
                        indirilenTumM3uIcerigi.append(bodyString).append("\n")
                        internettenYuklendiMi = true
                    }
                } catch (e: Exception) { e.printStackTrace() }
            }

            if (ulkeKodu == "tr") {
                val radyoKaynaklari = listOf(
                    "https://raw.githubusercontent.com/iprd-org/iprd/refs/heads/main/streams/tr/tr.m3u",
                    "https://raw.githubusercontent.com/junguler/m3u-radio-music-playlists/refs/heads/main/all-online_radio/turkey.m3u",
                    "https://raw.githubusercontent.com/hayatiptv/iptv/refs/heads/master/Radyo.m3u",
                    "https://raw.githubusercontent.com/hayatiptv/iptv/refs/heads/master/radyo-01.m3u"
                )

                for (radyoUrl in radyoKaynaklari) {
                    try {
                        val request = Request.Builder().url(radyoUrl).build()
                        val response = client.newCall(request).execute()
                        if (response.isSuccessful) {
                            val radyoBody = response.body?.string() ?: ""
                            if (radyoBody.isNotEmpty() && (radyoBody.contains("#EXTINF") || radyoBody.contains("#EXTM3U"))) {
                                indirilenTumM3uIcerigi.append("\n").append(radyoBody)
                                internettenYuklendiMi = true
                            }
                        }
                    } catch (e: Exception) { e.printStackTrace() }
                }
            }

            if (internettenYuklendiMi && indirilenTumM3uIcerigi.isNotEmpty()) {
                try {
                    context.openFileOutput(yedekDosyaAdi, Context.MODE_PRIVATE).use { output ->
                        output.write(indirilenTumM3uIcerigi.toString().toByteArray())
                    }
                } catch (e: Exception) { e.printStackTrace() }

                parseM3uContent(indirilenTumM3uIcerigi.toString(), parsedChannels)
            } else {
                try {
                    val file = context.getFileStreamPath(yedekDosyaAdi)
                    if (file.exists()) {
                        val yerelIcerik = context.openFileInput(yedekDosyaAdi).bufferedReader().use { it.readText() }
                        if (yerelIcerik.isNotEmpty()) {
                            parseM3uContent(yerelIcerik, parsedChannels)
                            withContext(Dispatchers.Main) {
                                Toast.makeText(context, "🌐 Çevrimdışı Mod: Kanallar yerel yedekten yüklendi.", Toast.LENGTH_LONG).show()
                            }
                        }
                    }
                } catch (ex: Exception) { ex.printStackTrace() }
            }

            withContext(Dispatchers.Main) {
                allChannelsState.clear()
                allChannelsState.addAll(parsedChannels)
                isLoading.value = false

                if (parsedChannels.isEmpty()) {
                    Toast.makeText(context, "⚠️ Kanal listesi indirilemedi!", Toast.LENGTH_LONG).show()
                } else {
                    val tvCount = parsedChannels.count { !it.isRadio }
                    val radioCount = parsedChannels.count { it.isRadio }
                    val ulkeAdi = countriesList.find { it.code == ulkeKodu }?.displayName ?: ulkeKodu.uppercase()
                    Toast.makeText(context, "🌍 $ulkeAdi: $tvCount TV ve $radioCount Radyo hazır.", Toast.LENGTH_SHORT).show()
                    restoreLastChannel(parsedChannels)
                }
            }
        }
    }

    private fun parseM3uContent(m3uContent: String, parsedChannels: MutableList<MediaChannel>) {
        var currentTitle = ""
        var currentLogo = ""

        m3uContent.lineSequence().forEach { line ->
            val trimmed = line.trim()
            if (trimmed.startsWith("#EXTINF:")) {
                currentLogo = if (trimmed.contains("tvg-logo=\"")) trimmed.substringAfter("tvg-logo=\"").substringBefore("\"") else ""
                currentTitle = trimmed.substringAfter(",").trim()
            } else if (trimmed.startsWith("http")) {
                if (currentTitle.isNotEmpty()) {
                    val upperTitle = currentTitle.uppercase()

                    val isRadio = trimmed.contains("radio") || upperTitle.contains("RADYO") || upperTitle.contains("FM") || upperTitle.contains("AUDIO")
                    val existingChannel = parsedChannels.find { it.title.uppercase() == upperTitle && it.isRadio == isRadio }

                    if (existingChannel != null) {
                        if (!existingChannel.urls.contains(trimmed)) {
                            (existingChannel.urls as MutableList<String>).add(trimmed)
                        }
                    } else {
                        val kat = when {
                            isRadio -> {
                                when {
                                    upperTitle.contains("HABER") || upperTitle.contains("NEWS") || upperTitle.contains("NTV") || upperTitle.contains("TRT RADYO 1") || upperTitle.contains("VRE") -> "Haber"
                                    upperTitle.contains("SPOR") || upperTitle.contains("SPORT") || upperTitle.contains("ASPOR") -> "Spor"
                                    else -> "Müzik"
                                }
                            }
                            upperTitle.contains("HABER") || upperTitle.contains("NEWS") || upperTitle.contains("24") -> "Haber"
                            upperTitle.contains("SPOR") || upperTitle.contains("SPORT") || upperTitle.contains("MATCH") -> "Spor"
                            upperTitle.contains("BELGESEL") || upperTitle.contains("DOCUMENTARY") || upperTitle.contains("NAT GEO") -> "Belgesel"
                            upperTitle.contains("ÇOCUK") || upperTitle.contains("KIDS") || upperTitle.contains("CARTOON") || upperTitle.contains("TRT ÇOCUK") -> "Çocuk"
                            else -> "Genel"
                        }

                        val prefix = if (isRadio) "radio_" else "tv_"
                        val uniqueId = "$prefix${trimmed.hashCode()}"

                        parsedChannels.add(MediaChannel(uniqueId, currentTitle, mutableListOf(trimmed), currentLogo, isRadio, kat))
                    }
                }
                currentTitle = ""; currentLogo = ""
            }
        }
    }

    override fun onPause() {
        super.onPause()
        val seciliKanal = currentChannelState.value
        if (!isInPictureInPictureMode) {
            if (seciliKanal == null || !seciliKanal.isRadio) {
                exoPlayer?.pause()
            }
        }
    }

    override fun onDestroy() { super.onDestroy(); exoPlayer?.release(); exoPlayer = null }
